def loop_size(node):
    turtle = hare = node
    hare = hare.next.next
    turtle = turtle.next
    while hare != turtle:
        hare = hare.next.next
        turtle = turtle.next
    hare = hare.next.next  
    turtle = turtle.next
    cnt = 1
